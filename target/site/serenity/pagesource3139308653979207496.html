    function parseIndonesianDate(dateStr) {

        const bulan = {

            'Januari': '01',

            'Februari': '02',

            'Maret': '03',

            'April': '04',

            'Mei': '05',

            'Juni': '06',

            'Juli': '07',

            'Agustus': '08',

            'September': '09',

            'Oktober': '10',

            'November': '11',

            'Desember': '12'

        };



        const parts = dateStr.split(' '); // ['22', 'Mei', '2025', '08:44']

        if (parts.length < 4) return null;



        const day = parts[0].padStart(2, '0');

        const month = bulan[parts[1]];

        const year = parts[2];

        const time = parts[3];



        if (!month) return null;



        return new Date(`${year}-${month}-${day}T${time}`);

    }



    function formatDate(data, indonesianFormat = true) {

        const date = indonesianFormat ? parseIndonesianDate(data) : new Date(data);
