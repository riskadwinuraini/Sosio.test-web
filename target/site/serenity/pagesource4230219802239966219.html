    function parseIndonesianDate(dateStr) {

        const bulan = {

            'Januari': '01',

            'Februari': '02',

            'Maret': '03',

            'April': '04',

            'Mei': '05',

            'Juni': '06',

            'Juli': '07',

            'Agustus': '08',

            'September': '09',

            'Oktober': '10',

            'November': '11',

            'Desember': '12'

        };



        const parts = dateStr.split(' '); // ['22', 'Mei', '2025', '08:44']

        if (parts.length < 4) return null;



        const day = parts[0].padStart(2, '0');

        const month = bulan[parts[1]];

        const year = parts[2];

        const time = parts[3];



        if (!month) return null;



        return new Date(`${year}-${month}-${day}T${time}`);

    }



    function formatDate(data, indonesianFormat = true) {

        const date = indonesianFormat ? parseIndonesianDate(data) : new Date(data);

        const options = {

            day: '2-digit',

            month: 'long', // show full Indonesian month (e.g., "Mei")

            year: 'numeric',

            hour: '2-digit',

            minute: '2-digit',

            hour12: false

        };

        const formatted = new Intl.DateTimeFormat(locale, options).format(date);

        return `${formatted} WIB`;

    }



    function formatSimpleDate(data, indonesianFormat = true) {

        const date = indonesianFormat ? parseIndonesianDate(data) : new Date(data);

    }

        function getFilterParams() {

            const inputs = $('input[name="kategori[]"]');

            inputs.each(function(i) {

                if ($(this).is(':checked')) {

                    kategoriParams += 'kategori[]='+$(this).attr('id');

                    if ((i + 1) < inputs.length) {

                        kategoriParams += '&';

                    }

                }

            });

            const form = $('#filterModal form');



            // Group checked checkboxes by name

            const checkedCheckboxes = {};

            form.find('input[type="checkbox"]:checked').each(function () {

                const name = this.name.replace(/\[\]$/, ''); // Remove brackets (e.g., f_sentimen_isu[])

                if (!checkedCheckboxes[name]) {

                    checkedCheckboxes[name] = []; // Initialize array if not exists

                }

                checkedCheckboxes[name].push(this.value);

            });



            // Get all selected values from select elements

            const selectedOptions = {};

            form.find('select').each(function () {

                const name = this.name;

                const value = $(this).val(); // Can be array if multiple

                if (value) {

                    selectedOptions[name] = value;

                }

            });

            // Merge both objects

            const filters = { ...checkedCheckboxes, ...selectedOptions };

            

            formFilterValues = filters;

            let periodeId = "40";

            const periodVal = $('#f_period').val();

            if (periodVal !== '' && periodVal !== '-Pilih-') {

                periodeId = periodVal;

            }

            formFilterValues.f_period = periodeId;



            // Stakeholder, profiling and profiling change status filter

            let fStakeholderStatus = $('#f_stakeholder_status').val();

            if (fStakeholderStatus !== '') {

                formFilterValues.f_stakeholder_status = fStakeholderStatus;

            }

            let fProfilingStatus = $('#f_profiling_status').val();

            if (fProfilingStatus !== '') {

                formFilterValues.f_profiling_status = fProfilingStatus;

            }

            let fProfilingChangeStatus = $('#f_profiling_change_status').val();

            if (fProfilingChangeStatus !== '') {

                formFilterValues.f_profiling_change_status = fProfilingChangeStatus;

            }

            return formFilterValues;

        }



        function buildQueryParams(params) {

            const query = [];



            for (const key in params) {

                const value = params[key];



                if (Array.isArray(value)) {

                    value.forEach(val => {

                        if (val !== null && val !== undefined && val !== '') {

                            query.push(`${encodeURIComponent(key)}[]=${encodeURIComponent(val)}`);

                        }

                    });

                } else if (value !== null && value !== undefined && value !== '') {

                    query.push(`${encodeURIComponent(key)}=${encodeURIComponent(value)}`);

                }

            }



            return query.join('&');

        }



        function countFilterTotal() {

            // Count checked checkboxes

            const checkboxTotal = $('#filterModal input[type="checkbox"]:checked').length;



            // Count select2 with selected values

            let selectTotal = 0;

            $('#filterModal select.select2').each(function () {

                const val = $(this).val();

                if (val && val !== '' && val.length > 0) {

                    selectTotal++;

                }

            });



            // Sum both totals

            const total = checkboxTotal + selectTotal;



            // Update display

            $('.total-filter-filled').html(total > 0 ? total : '');

        }

        function formatTanggalRange(date1, date2) {

            const months = [

                "Januari", "Februari", "Maret", "April", "Mei", "Juni",

                "Juli", "Agustus", "September", "Oktober", "November", "Desember"

            ];



            const formatDate = (date) => {

                const parts = date.split("-");

                const newDate = new Date(parts[0], parts[1] - 1, parts[2]);

                const day = newDate.getDate();

                const month = months[newDate.getMonth()];

                const year = newDate.getFullYear();

                return { day, month, year };

            };



            const dateObj1 = formatDate(date1);

            const dateObj2 = formatDate(date2);



            if (dateObj1.year === dateObj2.year) {

                return `${dateObj1.day} ${dateObj1.month} - ${dateObj2.day} ${dateObj2.month} ${dateObj2.year}`;

            } else {

                return `${dateObj1.day} ${dateObj1.month} ${dateObj1.year} - ${dateObj2.day} ${dateObj2.month} ${dateObj2.year}`;

            }

        }
